73  std::unique_ptr<ExpressionNode> Start(const gd::String &type,
75  const gd::String &objectName = "") {
76  size_t expressionStartPosition = GetCurrentPosition();
77  auto expression = Expression(type, objectName);
78 
79  const gd::String &inferredType = expression->type;
80 
81  // Check for extra characters at the end of the expression
82  if (!IsEndReached()) {
83  auto op = gd::make_unique<OperatorNode>(inferredType, ' ');
84  op->leftHandSide = std::move(expression);
85  op->rightHandSide = ReadUntilEnd("unknown");
86 
87  op->rightHandSide->diagnostic = RaiseSyntaxError(
88  _("The expression has extra character at the end that should be "
89  "removed (or completed if your expression is not finished)."));
90 
91  op->location = ExpressionParserLocation(expressionStartPosition,
92  GetCurrentPosition());
93  return std::move(op);
94  }
95 
96  return expression;
97  }

99  std::unique_ptr<ExpressionNode> Expression(
100  const gd::String &type, const gd::String &objectName = "") {
101  SkipAllWhitespaces();
102 
103  size_t expressionStartPosition = GetCurrentPosition();
104  std::unique_ptr<ExpressionNode> leftHandSide = Term(type, objectName);
105 
106  const gd::String &inferredType = leftHandSide->type;
107 
108  SkipAllWhitespaces();
109 
110  if (IsEndReached()) return leftHandSide;
111  if (CheckIfChar(IsExpressionEndingChar)) return leftHandSide;
112  if (CheckIfChar(IsExpressionOperator)) {
113  auto op = gd::make_unique<OperatorNode>(inferredType, GetCurrentChar());
114  op->leftHandSide = std::move(leftHandSide);
115  op->diagnostic = ValidateOperator(inferredType, GetCurrentChar());
116  SkipChar();
117  op->rightHandSide = Expression(inferredType, objectName);
118 
119  op->location = ExpressionParserLocation(expressionStartPosition,
120  GetCurrentPosition());
121  return std::move(op);
122  }
123 
124  if (inferredType == "string") {
125  leftHandSide->diagnostic = RaiseSyntaxError(
126  "You must add the operator + between texts or expressions. For "
127  "example: \"Your name: \" + VariableString(PlayerName).");
128  } else if (inferredType == "number") {
129  leftHandSide->diagnostic = RaiseSyntaxError(
130  "No operator found. Did you forget to enter an operator (like +, -, "
131  "* or /) between numbers or expressions?");
132  } else {
133  leftHandSide->diagnostic = RaiseSyntaxError(
134  "More than one term was found. Verify that your expression is "
135  "properly written.");
136  }
137 
138  auto op = gd::make_unique<OperatorNode>(inferredType, ' ');
139  op->leftHandSide = std::move(leftHandSide);
140  op->rightHandSide = Expression(inferredType, objectName);
141  op->location =
142  ExpressionParserLocation(expressionStartPosition, GetCurrentPosition());
143  return std::move(op);
144  }

146  std::unique_ptr<ExpressionNode> Term(const gd::String &type,
147  const gd::String &objectName) {
148  SkipAllWhitespaces();
149 
150  size_t expressionStartPosition = GetCurrentPosition();
151  std::unique_ptr<ExpressionNode> factor = Factor(type, objectName);
152 
153  const gd::String &inferredType = factor->type;
154 
155  SkipAllWhitespaces();
156 
157  // This while loop is used instead of a recursion (like in Expression)
158  // to guarantee the proper operator precedence. (Expression could also
159  // be reworked to use a while loop).
160  while (CheckIfChar(IsTermOperator)) {
161  auto op = gd::make_unique<OperatorNode>(inferredType, GetCurrentChar());
162  op->leftHandSide = std::move(factor);
163  op->diagnostic = ValidateOperator(inferredType, GetCurrentChar());
164  SkipChar();
165  op->rightHandSide = Factor(inferredType, objectName);
166  op->location = ExpressionParserLocation(expressionStartPosition,
167  GetCurrentPosition());
168  SkipAllWhitespaces();
169 
170  factor = std::move(op);
171  }
172 
173  return factor;
174  };

176  std::unique_ptr<ExpressionNode> Factor(const gd::String &type,
177  const gd::String &objectName) {
178  SkipAllWhitespaces();
179  size_t expressionStartPosition = GetCurrentPosition();
180 
181  if (CheckIfChar(IsQuote)) {
182  std::unique_ptr<ExpressionNode> factor = ReadText();
183  if (type == "number")
184  factor->diagnostic =
185  RaiseTypeError(_("You entered a text, but a number was expected."),
186  expressionStartPosition);
187  else if (type != "string" && type != "number|string")
188  factor->diagnostic = RaiseTypeError(
189  _("You entered a text, but this type was expected:") + type,
190  expressionStartPosition);
191 
192  return factor;
193  } else if (CheckIfChar(IsUnaryOperator)) {
194  auto unaryOperatorCharacter = GetCurrentChar();
195  SkipChar();
196 
197  auto operatorOperand = Factor(type, objectName);
198  const gd::String &inferredType = operatorOperand->type;
199 
200  auto unaryOperator = gd::make_unique<UnaryOperatorNode>(
201  inferredType, unaryOperatorCharacter);
202  unaryOperator->diagnostic = ValidateUnaryOperator(
203  inferredType, unaryOperatorCharacter, expressionStartPosition);
204  unaryOperator->factor = std::move(operatorOperand);
205  unaryOperator->location = ExpressionParserLocation(
206  expressionStartPosition, GetCurrentPosition());
207 
208  return std::move(unaryOperator);
209  } else if (CheckIfChar(IsNumberFirstChar)) {
210  std::unique_ptr<ExpressionNode> factor = ReadNumber();
211  if (type == "string")
212  factor->diagnostic = RaiseTypeError(
213  _("You entered a number, but a text was expected (in quotes)."),
214  expressionStartPosition);
215  else if (type != "number" && type != "number|string")
216  factor->diagnostic = RaiseTypeError(
217  _("You entered a number, but this type was expected:") + type,
218  expressionStartPosition);
219 
220  return factor;
221  } else if (CheckIfChar(IsOpeningParenthesis)) {
222  SkipChar();
223  std::unique_ptr<ExpressionNode> factor = SubExpression(type, objectName);
224 
225  if (!CheckIfChar(IsClosingParenthesis)) {
226  factor->diagnostic =
227  RaiseSyntaxError(_("Missing a closing parenthesis. Add a closing "
228  "parenthesis for each opening parenthesis."));
229  }
230  SkipIfChar(IsClosingParenthesis);
231  return factor;
232  } else if (IsIdentifierAllowedChar()) {
233  // This is a place where the grammar differs according to the
234  // type being expected.
235  if (gd::ParameterMetadata::IsExpression("variable", type)) {
236  return Variable(type, objectName);
237  } else {
238  return Identifier(type);
239  }
240  }
241 
242  std::unique_ptr<ExpressionNode> factor = ReadUntilWhitespace(type);
243  factor->diagnostic = RaiseEmptyError(type, expressionStartPosition);
244  return factor;
245  }

247  std::unique_ptr<SubExpressionNode> SubExpression(
248  const gd::String &type, const gd::String &objectName) {
249  size_t expressionStartPosition = GetCurrentPosition();

251  auto expression = Expression(type, objectName);
252  const gd::String &inferredType = expression->type;
253 
254  auto subExpression =
255  gd::make_unique<SubExpressionNode>(inferredType, std::move(expression));
256  subExpression->location =
257  ExpressionParserLocation(expressionStartPosition, GetCurrentPosition());
258 
259  return std::move(subExpression);

262  std::unique_ptr<IdentifierOrFunctionCallOrObjectFunctionNameOrEmptyNode>
263  Identifier(const gd::String &type) {
264  auto identifierAndLocation = ReadIdentifierName();
265  gd::String name = identifierAndLocation.name;
266  auto nameLocation = identifierAndLocation.location;
267 
268  SkipAllWhitespaces();
269 
270  // We consider a namespace separator to be allowed here and be part of the
271  // function name (or object name, but object names are not allowed to
272  // contain a ":"). This is because functions from extensions have their
273  // extension name prefix, and separated by the namespace separator. This
274  // could maybe be refactored to create different nodes in the future.
275  if (IsNamespaceSeparator()) {
276  SkipNamespaceSeparator();
277  SkipAllWhitespaces();
278 
279  auto postNamespaceIdentifierAndLocation = ReadIdentifierName();
280  name += NAMESPACE_SEPARATOR;
281  name += postNamespaceIdentifierAndLocation.name;
282  ExpressionParserLocation completeNameLocation(
283  nameLocation.GetStartPosition(),
284  postNamespaceIdentifierAndLocation.location.GetEndPosition());
285  nameLocation = completeNameLocation;
286  }
287 
288  if (CheckIfChar(IsOpeningParenthesis)) {
289  ExpressionParserLocation openingParenthesisLocation = SkipChar();
290  return FreeFunction(type, name, nameLocation, openingParenthesisLocation);
291  } else if (CheckIfChar(IsDot)) {
292  ExpressionParserLocation dotLocation = SkipChar();
293  SkipAllWhitespaces();
294  return ObjectFunctionOrBehaviorFunction(
295  type, name, nameLocation, dotLocation);
296  } else {
297  auto identifier = gd::make_unique<IdentifierNode>(name, type);
298  if (type == "string") {
299  identifier->diagnostic =
300  RaiseTypeError(_("You must wrap your text inside double quotes "
301  "(example: \"Hello world\")."),
302  nameLocation.GetStartPosition());
303  } else if (type == "number") {
304  identifier->diagnostic = RaiseTypeError(
305  _("You must enter a number."), nameLocation.GetStartPosition());
306  } else if (type == "number|string") {
307  identifier->diagnostic = RaiseTypeError(
308  _("You must enter a number or a text, wrapped inside double quotes "
309  "(example: \"Hello world\")."),
310  nameLocation.GetStartPosition());
311  } else if (!gd::ParameterMetadata::IsObject(type)) {
312  identifier->diagnostic = RaiseTypeError(
313  _("You've entered a name, but this type was expected:") + type,
314  nameLocation.GetStartPosition());
315  }
316 
317  identifier->location = ExpressionParserLocation(
318  nameLocation.GetStartPosition(), GetCurrentPosition());
319  return std::move(identifier);
320  }
321  }

338  std::unique_ptr<VariableAccessorOrVariableBracketAccessorNode>
339  VariableAccessorOrVariableBracketAccessor() {
340  size_t childStartPosition = GetCurrentPosition();
341 
342  SkipAllWhitespaces();
343  if (CheckIfChar(IsOpeningSquareBracket)) {
344  SkipChar();
345  auto child = gd::make_unique<VariableBracketAccessorNode>(
346  Expression("number|string"));
347 
348  if (!CheckIfChar(IsClosingSquareBracket)) {
349  child->diagnostic =
350  RaiseSyntaxError(_("Missing a closing bracket. Add a closing "
351  "bracket for each opening bracket."));
352  }
353  SkipIfChar(IsClosingSquareBracket);
354  child->child = VariableAccessorOrVariableBracketAccessor();
355  child->location =
356  ExpressionParserLocation(childStartPosition, GetCurrentPosition());
357 
358  return std::move(child);
359  } else if (CheckIfChar(IsDot)) {
360  auto dotLocation = SkipChar();
361  SkipAllWhitespaces();
362 
363  auto identifierAndLocation = ReadIdentifierName();
364  auto child =
365  gd::make_unique<VariableAccessorNode>(identifierAndLocation.name);
366  child->child = VariableAccessorOrVariableBracketAccessor();
367  child->nameLocation = identifierAndLocation.location;
368  child->dotLocation = dotLocation;
369  child->location =
370  ExpressionParserLocation(childStartPosition, GetCurrentPosition());
371 
372  return std::move(child);
373  }
374 
375  return std::move(
376  std::unique_ptr<VariableAccessorOrVariableBracketAccessorNode>());
377  }

379  std::unique_ptr<FunctionCallNode> FreeFunction(
380  const gd::String &type,
381  const gd::String &functionFullName,
382  const ExpressionParserLocation &identifierLocation,
383  const ExpressionParserLocation &openingParenthesisLocation) {
384  // TODO: error if trying to use function for type != "number" && != "string"
385  // + Test for it
386 
387  const gd::ExpressionMetadata &metadata =
388  MetadataProvider::GetAnyExpressionMetadata(platform, functionFullName);
389 
390  auto parametersNode = Parameters(metadata.parameters);
391  auto function =
392  gd::make_unique<FunctionCallNode>(metadata.GetReturnType(),
393  std::move(parametersNode.parameters),
394  metadata,
395  functionFullName);
396  function->diagnostic = std::move(parametersNode.diagnostic);
397  if (!function->diagnostic) // TODO: reverse the order of diagnostic?
398  function->diagnostic = ValidateFunction(
399  type, *function, identifierLocation.GetStartPosition());
400 
401  function->location = ExpressionParserLocation(
402  identifierLocation.GetStartPosition(), GetCurrentPosition());
403  function->functionNameLocation = identifierLocation;
404  function->openingParenthesisLocation = openingParenthesisLocation;
405  function->closingParenthesisLocation =
406  parametersNode.closingParenthesisLocation;
407 
408  return std::move(function);
409  }

413  const gd::String &type,
414  const gd::String &objectName,
415  const ExpressionParserLocation &objectNameLocation,
416  const ExpressionParserLocation &objectNameDotLocation) {

419  const auto &objectFunctionOrBehaviorNameLocation =

422  SkipAllWhitespaces();

448  objectName,

454  function->diagnostic = ValidateFunction(
455  type, *function, objectNameLocation.GetStartPosition());
456 
457  function->location = ExpressionParserLocation(
458  objectNameLocation.GetStartPosition(), GetCurrentPosition());
459  function->objectNameLocation = objectNameLocation;
460  function->objectNameDotLocation = objectNameDotLocation;
461  function->functionNameLocation = objectFunctionOrBehaviorNameLocation;
462  function->openingParenthesisLocation = openingParenthesisLocation;
463  function->closingParenthesisLocation =

469  type, objectName, objectFunctionOrBehaviorName);
470  node->diagnostic = RaiseSyntaxError(
471  _("An opening parenthesis (for an object expression), or double colon "

473 
474  node->location = ExpressionParserLocation(
475  objectNameLocation.GetStartPosition(), GetCurrentPosition());
476  node->objectNameLocation = objectNameLocation;
477  node->objectNameDotLocation = objectNameDotLocation;
478  node->objectFunctionOrBehaviorNameLocation =
479  objectFunctionOrBehaviorNameLocation;
480  return std::move(node);

551  // A temporary node that will be integrated into function nodes.
552  struct ParametersNode {
553  std::vector<std::unique_ptr<ExpressionNode>> parameters;
554  std::unique_ptr<gd::ExpressionParserError> diagnostic;
555  ExpressionParserLocation closingParenthesisLocation;
556  };
557 
558  ParametersNode Parameters(
559  std::vector<gd::ParameterMetadata> parameterMetadata,
560  const gd::String &objectName = "",
561  const gd::String &behaviorName = "") {
562  std::vector<std::unique_ptr<ExpressionNode>> parameters;
563  gd::String lastObjectName = "";
564 
565  // By convention, object is always the first parameter, and behavior the
566  // second one.
567  size_t parameterIndex =
568  WrittenParametersFirstIndex(objectName, behaviorName);
569 
570  while (!IsEndReached()) {
571  SkipAllWhitespaces();
572 
573  if (CheckIfChar(IsClosingParenthesis)) {
574  auto closingParenthesisLocation = SkipChar();
575  return ParametersNode{
576  std::move(parameters), nullptr, closingParenthesisLocation};
577  } else {
578  if (parameterIndex < parameterMetadata.size()) {
579  const gd::String &type = parameterMetadata[parameterIndex].GetType();
580  if (parameterMetadata[parameterIndex].IsCodeOnly()) {
581  // Do nothing, code only parameters are not written in expressions.
582  } else if (gd::ParameterMetadata::IsExpression("number", type)) {
583  parameters.push_back(Expression("number"));
584  } else if (gd::ParameterMetadata::IsExpression("string", type)) {
585  parameters.push_back(Expression("string"));
586  } else if (gd::ParameterMetadata::IsExpression("variable", type)) {
587  parameters.push_back(Expression(
588  type, lastObjectName.empty() ? objectName : lastObjectName));
589  } else if (gd::ParameterMetadata::IsObject(type)) {
590  size_t parameterStartPosition = GetCurrentPosition();
591  std::unique_ptr<ExpressionNode> objectExpression = Expression(type);
592 
593  // Memorize the last object name. By convention, parameters that
594  // require an object (mainly, "objectvar" and "behavior") should be
595  // placed after the object in the list of parameters (if possible,
596  // just after). Search "lastObjectName" in the codebase for other
597  // place where this convention is enforced.
598  if (auto identifierNode =
599  dynamic_cast<IdentifierNode *>(objectExpression.get())) {
600  lastObjectName = identifierNode->identifierName;
601  } else {
602  objectExpression->diagnostic =
603  gd::make_unique<ExpressionParserError>(
604  "malformed_object_parameter",
605  _("An object name was expected but something else was "
606  "written. Enter just the name of the object for this "
607  "parameter."),
608  parameterStartPosition,
609  GetCurrentPosition());
610  }
611 
612  parameters.push_back(std::move(objectExpression));
613  } else {
614  size_t parameterStartPosition = GetCurrentPosition();
615  parameters.push_back(Expression("unknown"));
616  parameters.back()->diagnostic =
617  gd::make_unique<ExpressionParserError>(
618  "unknown_parameter_type",
619  _("This function is improperly set up. Reach out to the "
620  "extension developer or a GDevelop maintainer to fix "
621  "this issue"),
622  parameterStartPosition,
623  GetCurrentPosition());
624  }
625  } else {
626  size_t parameterStartPosition = GetCurrentPosition();
627  parameters.push_back(Expression("unknown"));
628  parameters.back()
629  ->diagnostic = gd::make_unique<ExpressionParserError>(
630  "extra_parameter",
631  _("This parameter was not expected by this expression. Remove it "
632  "or verify that you've entered the proper expression name."),
633  parameterStartPosition,
634  GetCurrentPosition());
635  }
636 
637  SkipAllWhitespaces();
638  SkipIfChar(IsParameterSeparator);
639  parameterIndex++;
640  }
641  }
642 
643  ExpressionParserLocation invalidClosingParenthesisLocation;
644  return ParametersNode{
645  std::move(parameters),
646  RaiseSyntaxError(_("The list of parameters is not terminated. Add a "
647  "closing parenthesis to end the parameters.")),
648  invalidClosingParenthesisLocation};
649  }

661  std::unique_ptr<ExpressionParserDiagnostic> ValidateOperator(
662  const gd::String &type, gd::String::value_type operatorChar) {
663  if (type == "number") {
664  if (operatorChar == '+' || operatorChar == '-' || operatorChar == '/' ||
665  operatorChar == '*') {
666  return gd::make_unique<ExpressionParserDiagnostic>();
667  }
668 
669  return gd::make_unique<ExpressionParserError>(
670  "invalid_operator",
671  _("You've used an operator that is not supported. Operator should be "
672  "either +, -, / or *."),
673  GetCurrentPosition());
674  } else if (type == "string") {
675  if (operatorChar == '+') {
676  return gd::make_unique<ExpressionParserDiagnostic>();
677  }
678 
679  return gd::make_unique<ExpressionParserError>(
680  "invalid_operator",
681  _("You've used an operator that is not supported. Only + can be used "
682  "to concatenate texts."),
683  GetCurrentPosition());
684  } else if (gd::ParameterMetadata::IsObject(type)) {
685  return gd::make_unique<ExpressionParserError>(
686  "invalid_operator",
687  _("Operators (+, -, /, *) can't be used with an object name. Remove "
688  "the operator."),
689  GetCurrentPosition());
690  } else if (gd::ParameterMetadata::IsExpression("variable", type)) {
691  return gd::make_unique<ExpressionParserError>(
692  "invalid_operator",
693  _("Operators (+, -, /, *) can't be used in variable names. Remove "
694  "the operator from the variable name."),
695  GetCurrentPosition());
696  }
697 
698  return gd::make_unique<ExpressionParserDiagnostic>();
699  }

904  while (lastCharacterPos < name.size() &&
905  IsWhitespace(name[lastCharacterPos])) {
906  lastCharacterPos--;
907  }

909  name.erase(lastCharacterPos + 1);

916  // inside the identifier are allowed for compatibility).
