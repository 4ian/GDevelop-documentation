79  const gd::String &behaviorName) {

89  std::unique_ptr<ExpressionNode> Start(const gd::String &type,
91  const gd::String &objectName = "") {
92  size_t expressionStartPosition = GetCurrentPosition();
93  auto expression = Expression(type, objectName);
94 
95  const gd::String &inferredType = expression->type;
96 
97  // Check for extra characters at the end of the expression
98  if (!IsEndReached()) {
99  auto op = gd::make_unique<OperatorNode>(inferredType, ' ');
100  op->leftHandSide = std::move(expression);
101  op->rightHandSide = ReadUntilEnd("unknown");
102 
103  op->rightHandSide->diagnostic = RaiseSyntaxError(
104  _("The expression has extra character at the end that should be "
105  "removed (or completed if your expression is not finished)."));
106 
107  op->location = ExpressionParserLocation(expressionStartPosition,
108  GetCurrentPosition());
109  return std::move(op);
110  }
111 
112  return expression;
113  }

115  std::unique_ptr<ExpressionNode> Expression(
116  const gd::String &type, const gd::String &objectName = "") {
117  SkipAllWhitespaces();
118 
119  size_t expressionStartPosition = GetCurrentPosition();
120  std::unique_ptr<ExpressionNode> leftHandSide = Term(type, objectName);
121 
122  const gd::String &inferredType = leftHandSide->type;
123 
124  SkipAllWhitespaces();
125 
126  if (IsEndReached()) return leftHandSide;
127  if (CheckIfChar(IsExpressionEndingChar)) return leftHandSide;
128  if (CheckIfChar(IsExpressionOperator)) {
129  auto op = gd::make_unique<OperatorNode>(inferredType, GetCurrentChar());
130  op->leftHandSide = std::move(leftHandSide);
131  op->diagnostic = ValidateOperator(inferredType, GetCurrentChar());
132  SkipChar();
133  op->rightHandSide = Expression(inferredType, objectName);
134 
135  op->location = ExpressionParserLocation(expressionStartPosition,
136  GetCurrentPosition());
137  return std::move(op);
138  }
139 
140  if (inferredType == "string") {
141  leftHandSide->diagnostic = RaiseSyntaxError(
142  "You must add the operator + between texts or expressions. For "
143  "example: \"Your name: \" + VariableString(PlayerName).");
144  } else if (inferredType == "number") {
145  leftHandSide->diagnostic = RaiseSyntaxError(
146  "No operator found. Did you forget to enter an operator (like +, -, "
147  "* or /) between numbers or expressions?");
148  } else {
149  leftHandSide->diagnostic = RaiseSyntaxError(
150  "More than one term was found. Verify that your expression is "
151  "properly written.");
152  }
153 
154  auto op = gd::make_unique<OperatorNode>(inferredType, ' ');
155  op->leftHandSide = std::move(leftHandSide);
156  op->rightHandSide = Expression(inferredType, objectName);
157  op->location =
158  ExpressionParserLocation(expressionStartPosition, GetCurrentPosition());
159  return std::move(op);
160  }

162  std::unique_ptr<ExpressionNode> Term(const gd::String &type,
163  const gd::String &objectName) {
164  SkipAllWhitespaces();
165 
166  size_t expressionStartPosition = GetCurrentPosition();
167  std::unique_ptr<ExpressionNode> factor = Factor(type, objectName);
168 
169  const gd::String &inferredType = factor->type;
170 
171  SkipAllWhitespaces();
172 
173  // This while loop is used instead of a recursion (like in Expression)
174  // to guarantee the proper operator precedence. (Expression could also
175  // be reworked to use a while loop).
176  while (CheckIfChar(IsTermOperator)) {
177  auto op = gd::make_unique<OperatorNode>(inferredType, GetCurrentChar());
178  op->leftHandSide = std::move(factor);
179  op->diagnostic = ValidateOperator(inferredType, GetCurrentChar());
180  SkipChar();
181  op->rightHandSide = Factor(inferredType, objectName);
182  op->location = ExpressionParserLocation(expressionStartPosition,
183  GetCurrentPosition());
184  SkipAllWhitespaces();
185 
186  factor = std::move(op);
187  }
188 
189  return factor;
190  };

192  std::unique_ptr<ExpressionNode> Factor(const gd::String &type,
193  const gd::String &objectName) {
194  SkipAllWhitespaces();
195  size_t expressionStartPosition = GetCurrentPosition();
196 
197  if (CheckIfChar(IsQuote)) {
198  std::unique_ptr<ExpressionNode> factor = ReadText();
199  if (type == "number")
200  factor->diagnostic =
201  RaiseTypeError(_("You entered a text, but a number was expected."),
202  expressionStartPosition);
203  else if (type != "string" && type != "number|string")
204  factor->diagnostic = RaiseTypeError(
205  _("You entered a text, but this type was expected:") + type,
206  expressionStartPosition);
207 
208  return factor;
209  } else if (CheckIfChar(IsUnaryOperator)) {
210  auto unaryOperatorCharacter = GetCurrentChar();
211  SkipChar();
212 
213  auto operatorOperand = Factor(type, objectName);
214  const gd::String &inferredType = operatorOperand->type;
215 
216  auto unaryOperator = gd::make_unique<UnaryOperatorNode>(
217  inferredType, unaryOperatorCharacter);
218  unaryOperator->diagnostic = ValidateUnaryOperator(
219  inferredType, unaryOperatorCharacter, expressionStartPosition);
220  unaryOperator->factor = std::move(operatorOperand);
221  unaryOperator->location = ExpressionParserLocation(
222  expressionStartPosition, GetCurrentPosition());
223 
224  return std::move(unaryOperator);
225  } else if (CheckIfChar(IsNumberFirstChar)) {
226  std::unique_ptr<ExpressionNode> factor = ReadNumber();
227  if (type == "string")
228  factor->diagnostic = RaiseTypeError(
229  _("You entered a number, but a text was expected (in quotes)."),
230  expressionStartPosition);
231  else if (type != "number" && type != "number|string")
232  factor->diagnostic = RaiseTypeError(
233  _("You entered a number, but this type was expected:") + type,
234  expressionStartPosition);
235 
236  return factor;
237  } else if (CheckIfChar(IsOpeningParenthesis)) {
238  SkipChar();
239  std::unique_ptr<ExpressionNode> factor = SubExpression(type, objectName);
240 
241  if (!CheckIfChar(IsClosingParenthesis)) {
242  factor->diagnostic =
243  RaiseSyntaxError(_("Missing a closing parenthesis. Add a closing "
244  "parenthesis for each opening parenthesis."));
245  }
246  SkipIfChar(IsClosingParenthesis);
247  return factor;
248  } else if (IsIdentifierAllowedChar()) {
249  // This is a place where the grammar differs according to the
250  // type being expected.
251  if (gd::ParameterMetadata::IsExpression("variable", type)) {
252  return Variable(type, objectName);
253  } else {
254  return Identifier(type);
255  }
256  }
257 
258  std::unique_ptr<ExpressionNode> factor = ReadUntilWhitespace(type);
259  factor->diagnostic = RaiseEmptyError(type, expressionStartPosition);
260  return factor;
261  }

264  const gd::String &type, const gd::String &objectName) {
265  size_t expressionStartPosition = GetCurrentPosition();

274 
275  return std::move(subExpression);

278  std::unique_ptr<IdentifierOrFunctionCallOrObjectFunctionNameOrEmptyNode>
279  Identifier(const gd::String &type) {
280  auto identifierAndLocation = ReadIdentifierName();
281  gd::String name = identifierAndLocation.name;
282  auto nameLocation = identifierAndLocation.location;
283 
284  SkipAllWhitespaces();
285 
286  // We consider a namespace separator to be allowed here and be part of the
287  // function name (or object name, but object names are not allowed to
288  // contain a ":"). This is because functions from extensions have their
289  // extension name prefix, and separated by the namespace separator. This
290  // could maybe be refactored to create different nodes in the future.
291  if (IsNamespaceSeparator()) {
292  SkipNamespaceSeparator();
293  SkipAllWhitespaces();
294 
295  auto postNamespaceIdentifierAndLocation = ReadIdentifierName();
296  name += NAMESPACE_SEPARATOR;
297  name += postNamespaceIdentifierAndLocation.name;
298  ExpressionParserLocation completeNameLocation(
299  nameLocation.GetStartPosition(),
300  postNamespaceIdentifierAndLocation.location.GetEndPosition());
301  nameLocation = completeNameLocation;
302  }
303 
304  if (CheckIfChar(IsOpeningParenthesis)) {
305  ExpressionParserLocation openingParenthesisLocation = SkipChar();
306  return FreeFunction(type, name, nameLocation, openingParenthesisLocation);
307  } else if (CheckIfChar(IsDot)) {
308  ExpressionParserLocation dotLocation = SkipChar();
309  SkipAllWhitespaces();
310  return ObjectFunctionOrBehaviorFunction(
311  type, name, nameLocation, dotLocation);
312  } else {
313  auto identifier = gd::make_unique<IdentifierNode>(name, type);
314  if (type == "string") {
315  identifier->diagnostic =
316  RaiseTypeError(_("You must wrap your text inside double quotes "
317  "(example: \"Hello world\")."),
318  nameLocation.GetStartPosition());
319  } else if (type == "number") {
320  identifier->diagnostic = RaiseTypeError(
321  _("You must enter a number."), nameLocation.GetStartPosition());
322  } else if (type == "number|string") {
323  identifier->diagnostic = RaiseTypeError(
324  _("You must enter a number or a text, wrapped inside double quotes "
325  "(example: \"Hello world\")."),
326  nameLocation.GetStartPosition());
327  } else if (!gd::ParameterMetadata::IsObject(type)) {
328  identifier->diagnostic = RaiseTypeError(
329  _("You've entered a name, but this type was expected:") + type,
330  nameLocation.GetStartPosition());
331  }
332 
333  identifier->location = ExpressionParserLocation(
334  nameLocation.GetStartPosition(), GetCurrentPosition());
335  return std::move(identifier);
336  }
337  }

354  std::unique_ptr<VariableAccessorOrVariableBracketAccessorNode>
355  VariableAccessorOrVariableBracketAccessor() {
356  size_t childStartPosition = GetCurrentPosition();
357 
358  SkipAllWhitespaces();
359  if (CheckIfChar(IsOpeningSquareBracket)) {
360  SkipChar();
361  auto child = gd::make_unique<VariableBracketAccessorNode>(
362  Expression("number|string"));
363 
364  if (!CheckIfChar(IsClosingSquareBracket)) {
365  child->diagnostic =
366  RaiseSyntaxError(_("Missing a closing bracket. Add a closing "
367  "bracket for each opening bracket."));
368  }
369  SkipIfChar(IsClosingSquareBracket);
370  child->child = VariableAccessorOrVariableBracketAccessor();
371  child->location =
372  ExpressionParserLocation(childStartPosition, GetCurrentPosition());
373 
374  return std::move(child);
375  } else if (CheckIfChar(IsDot)) {
376  auto dotLocation = SkipChar();
377  SkipAllWhitespaces();
378 
379  auto identifierAndLocation = ReadIdentifierName();
380  auto child =
381  gd::make_unique<VariableAccessorNode>(identifierAndLocation.name);
382  child->child = VariableAccessorOrVariableBracketAccessor();
383  child->nameLocation = identifierAndLocation.location;
384  child->dotLocation = dotLocation;
385  child->location =
386  ExpressionParserLocation(childStartPosition, GetCurrentPosition());
387 
388  return std::move(child);
389  }
390 
391  return std::move(
392  std::unique_ptr<VariableAccessorOrVariableBracketAccessorNode>());
393  }

395  std::unique_ptr<FunctionCallNode> FreeFunction(
396  const gd::String &type,
397  const gd::String &functionFullName,
398  const ExpressionParserLocation &identifierLocation,
399  const ExpressionParserLocation &openingParenthesisLocation) {
400  // TODO: error if trying to use function for type != "number" && != "string"
401  // + Test for it
402 
403  const gd::ExpressionMetadata &metadata =
404  MetadataProvider::GetAnyExpressionMetadata(platform, functionFullName);
405 
406  auto parametersNode = Parameters(metadata.parameters);
407  auto function =
408  gd::make_unique<FunctionCallNode>(metadata.GetReturnType(),
409  std::move(parametersNode.parameters),
410  metadata,
411  functionFullName);
412  function->diagnostic = std::move(parametersNode.diagnostic);
413  if (!function->diagnostic) // TODO: reverse the order of diagnostic?
414  function->diagnostic = ValidateFunction(

416 
417  function->location = ExpressionParserLocation(
418  identifierLocation.GetStartPosition(), GetCurrentPosition());
419  function->functionNameLocation = identifierLocation;
420  function->openingParenthesisLocation = openingParenthesisLocation;
421  function->closingParenthesisLocation =
422  parametersNode.closingParenthesisLocation;
423 
424  return std::move(function);

427  std::unique_ptr<FunctionCallOrObjectFunctionNameOrEmptyNode>
428  ObjectFunctionOrBehaviorFunction(
429  const gd::String &type,
430  const gd::String &objectName,
431  const ExpressionParserLocation &objectNameLocation,
432  const ExpressionParserLocation &objectNameDotLocation) {
433  auto identifierAndLocation = ReadIdentifierName();
434  const gd::String &objectFunctionOrBehaviorName = identifierAndLocation.name;
435  const auto &objectFunctionOrBehaviorNameLocation =
436  identifierAndLocation.location;
437 
438  SkipAllWhitespaces();
439 
440  if (IsNamespaceSeparator()) {
441  ExpressionParserLocation namespaceSeparatorLocation =
442  SkipNamespaceSeparator();
443  SkipAllWhitespaces();
444  return BehaviorFunction(type,
445  objectName,
446  objectFunctionOrBehaviorName,
447  objectNameLocation,
448  objectNameDotLocation,
449  objectFunctionOrBehaviorNameLocation,
450  namespaceSeparatorLocation);
451  } else if (CheckIfChar(IsOpeningParenthesis)) {
452  ExpressionParserLocation openingParenthesisLocation = SkipChar();
453 
454  gd::String objectType =
455  GetTypeOfObject(globalObjectsContainer, objectsContainer, objectName);
456 
457  const gd::ExpressionMetadata &metadata =
458  MetadataProvider::GetObjectAnyExpressionMetadata(
459  platform, objectType, objectFunctionOrBehaviorName);
460 
461  auto parametersNode = Parameters(metadata.parameters, objectName);
462  auto function = gd::make_unique<FunctionCallNode>(
463  metadata.GetReturnType(),
464  objectName,
465  std::move(parametersNode.parameters),
466  metadata,
467  objectFunctionOrBehaviorName);
468  function->diagnostic = std::move(parametersNode.diagnostic);
469  if (!function->diagnostic) // TODO: reverse the order of diagnostic?
470  function->diagnostic = ValidateFunction(
471  type, *function, objectNameLocation.GetStartPosition());
472 
473  function->location = ExpressionParserLocation(
474  objectNameLocation.GetStartPosition(), GetCurrentPosition());
475  function->objectNameLocation = objectNameLocation;
476  function->objectNameDotLocation = objectNameDotLocation;
477  function->functionNameLocation = objectFunctionOrBehaviorNameLocation;
478  function->openingParenthesisLocation = openingParenthesisLocation;
479  function->closingParenthesisLocation =
480  parametersNode.closingParenthesisLocation;
481  return std::move(function);
482  }
483 
484  auto node = gd::make_unique<ObjectFunctionNameNode>(
485  type, objectName, objectFunctionOrBehaviorName);
486  node->diagnostic = RaiseSyntaxError(
487  _("An opening parenthesis (for an object expression), or double colon "
488  "(::) was expected (for a behavior expression)."));
489 
490  node->location = ExpressionParserLocation(
491  objectNameLocation.GetStartPosition(), GetCurrentPosition());
492  node->objectNameLocation = objectNameLocation;
493  node->objectNameDotLocation = objectNameDotLocation;
494  node->objectFunctionOrBehaviorNameLocation =
495  objectFunctionOrBehaviorNameLocation;
496  return std::move(node);

567  // A temporary node that will be integrated into function nodes.
568  struct ParametersNode {
569  std::vector<std::unique_ptr<ExpressionNode>> parameters;
570  std::unique_ptr<gd::ExpressionParserError> diagnostic;
571  ExpressionParserLocation closingParenthesisLocation;
572  };
573 
574  ParametersNode Parameters(
575  std::vector<gd::ParameterMetadata> parameterMetadata,
576  const gd::String &objectName = "",
577  const gd::String &behaviorName = "") {
578  std::vector<std::unique_ptr<ExpressionNode>> parameters;
579  gd::String lastObjectName = "";
580 
581  // By convention, object is always the first parameter, and behavior the
582  // second one.
583  size_t parameterIndex =
584  WrittenParametersFirstIndex(objectName, behaviorName);
585 
586  while (!IsEndReached()) {
587  SkipAllWhitespaces();
588 
589  if (CheckIfChar(IsClosingParenthesis)) {
590  auto closingParenthesisLocation = SkipChar();
591  return ParametersNode{
592  std::move(parameters), nullptr, closingParenthesisLocation};
593  } else {
594  if (parameterIndex < parameterMetadata.size()) {
595  const gd::String &type = parameterMetadata[parameterIndex].GetType();
596  if (parameterMetadata[parameterIndex].IsCodeOnly()) {
597  // Do nothing, code only parameters are not written in expressions.
598  } else if (gd::ParameterMetadata::IsExpression("number", type)) {
599  parameters.push_back(Expression("number"));
600  } else if (gd::ParameterMetadata::IsExpression("string", type)) {
601  parameters.push_back(Expression("string"));
602  } else if (gd::ParameterMetadata::IsExpression("variable", type)) {
603  parameters.push_back(Expression(
604  type, lastObjectName.empty() ? objectName : lastObjectName));
605  } else if (gd::ParameterMetadata::IsObject(type)) {
606  size_t parameterStartPosition = GetCurrentPosition();
607  std::unique_ptr<ExpressionNode> objectExpression = Expression(type);
608 
609  // Memorize the last object name. By convention, parameters that
610  // require an object (mainly, "objectvar" and "behavior") should be
611  // placed after the object in the list of parameters (if possible,
612  // just after). Search "lastObjectName" in the codebase for other
613  // place where this convention is enforced.
614  if (auto identifierNode =
615  dynamic_cast<IdentifierNode *>(objectExpression.get())) {
616  lastObjectName = identifierNode->identifierName;
617  } else {
618  objectExpression->diagnostic =
619  gd::make_unique<ExpressionParserError>(
620  "malformed_object_parameter",
621  _("An object name was expected but something else was "
622  "written. Enter just the name of the object for this "
623  "parameter."),
624  parameterStartPosition,
625  GetCurrentPosition());
626  }
627 
628  parameters.push_back(std::move(objectExpression));
629  } else {
630  size_t parameterStartPosition = GetCurrentPosition();
631  parameters.push_back(Expression("unknown"));
632  parameters.back()->diagnostic =
633  gd::make_unique<ExpressionParserError>(
634  "unknown_parameter_type",
635  _("This function is improperly set up. Reach out to the "
636  "extension developer or a GDevelop maintainer to fix "
637  "this issue"),
638  parameterStartPosition,
639  GetCurrentPosition());
640  }
641  } else {
642  size_t parameterStartPosition = GetCurrentPosition();
643  parameters.push_back(Expression("unknown"));
644  parameters.back()
645  ->diagnostic = gd::make_unique<ExpressionParserError>(
646  "extra_parameter",
647  _("This parameter was not expected by this expression. Remove it "
648  "or verify that you've entered the proper expression name."),
649  parameterStartPosition,
650  GetCurrentPosition());
651  }
652 
653  SkipAllWhitespaces();
654  SkipIfChar(IsParameterSeparator);
655  parameterIndex++;
656  }
657  }
658 
659  ExpressionParserLocation invalidClosingParenthesisLocation;
660  return ParametersNode{
661  std::move(parameters),
662  RaiseSyntaxError(_("The list of parameters is not terminated. Add a "
663  "closing parenthesis to end the parameters.")),
664  invalidClosingParenthesisLocation};
665  }

677  std::unique_ptr<ExpressionParserDiagnostic> ValidateOperator(
678  const gd::String &type, gd::String::value_type operatorChar) {
679  if (type == "number") {
680  if (operatorChar == '+' || operatorChar == '-' || operatorChar == '/' ||
681  operatorChar == '*') {
682  return gd::make_unique<ExpressionParserDiagnostic>();
683  }
684 
685  return gd::make_unique<ExpressionParserError>(
686  "invalid_operator",
687  _("You've used an operator that is not supported. Operator should be "
688  "either +, -, / or *."),
689  GetCurrentPosition());
690  } else if (type == "string") {
691  if (operatorChar == '+') {
692  return gd::make_unique<ExpressionParserDiagnostic>();
693  }
694 
695  return gd::make_unique<ExpressionParserError>(
696  "invalid_operator",
697  _("You've used an operator that is not supported. Only + can be used "
698  "to concatenate texts."),
699  GetCurrentPosition());
700  } else if (gd::ParameterMetadata::IsObject(type)) {
701  return gd::make_unique<ExpressionParserError>(
702  "invalid_operator",
703  _("Operators (+, -, /, *) can't be used with an object name. Remove "
704  "the operator."),
705  GetCurrentPosition());
706  } else if (gd::ParameterMetadata::IsExpression("variable", type)) {
707  return gd::make_unique<ExpressionParserError>(
708  "invalid_operator",

921  IsWhitespace(name[lastCharacterPos])) {
922  lastCharacterPos--;
923  }

925  name.erase(lastCharacterPos + 1);

930  name,

933  ExpressionParserLocation(startPosition, startPosition + name.size())};
