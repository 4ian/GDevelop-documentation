74  void OnVisitOperatorNode(OperatorNode& node) override {
75  ReportAnyError(node);
76 
77  node.leftHandSide->Visit(*this);
78  const Type leftType = childType;
79 
80  if (leftType == Type::Number) {
81  if (node.op == ' ') {
82  RaiseError("syntax_error",
83  "No operator found. Did you forget to enter an operator (like +, -, "
84  "* or /) between numbers or expressions?", node.rightHandSide->location);
85  }
86  }
87  else if (leftType == Type::String) {
88  if (node.op == ' ') {
89  RaiseError("syntax_error",
90  "You must add the operator + between texts or expressions. For "
91  "example: \"Your name: \" + VariableString(PlayerName).", node.rightHandSide->location);
92  }
93  else if (node.op != '+') {
94  RaiseOperatorError(
95  _("You've used an operator that is not supported. Only + can be used "
96  "to concatenate texts."),
97  ExpressionParserLocation(node.leftHandSide->location.GetEndPosition() + 1, node.location.GetEndPosition()));
98  }
99  } else if (leftType == Type::Object) {
100  RaiseOperatorError(
101  _("Operators (+, -, /, *) can't be used with an object name. Remove "
102  "the operator."),
103  node.rightHandSide->location);
104  } else if (leftType == Type::Variable) {
105  RaiseOperatorError(
106  _("Operators (+, -, /, *) can't be used in variable names. Remove "
107  "the operator from the variable name."),
108  node.rightHandSide->location);
109  }
110 
111  parentType = leftType;
112  node.rightHandSide->Visit(*this);
113  const Type rightType = childType;
114 
115  childType = leftType;
116  }
117  void OnVisitUnaryOperatorNode(UnaryOperatorNode& node) override {
118  ReportAnyError(node);
119  node.factor->Visit(*this);
120  const Type rightType = childType;
121 
122  if (rightType == Type::Number) {
123  if (node.op != '+' && node.op != '-') {
124  // This is actually a dead code because the parser takes them as
125  // binary operations with an empty left side which makes as much sense.
126  RaiseTypeError(
127  _("You've used an \"unary\" operator that is not supported. Operator "
128  "should be "
129  "either + or -."),
130  node.location);
131  }
132  } else if (rightType == Type::String) {
133  RaiseTypeError(
134  _("You've used an operator that is not supported. Only + can be used "
135  "to concatenate texts, and must be placed between two texts (or "
136  "expressions)."),
137  node.location);
138  } else if (rightType == Type::Object) {
139  RaiseTypeError(
140  _("Operators (+, -) can't be used with an object name. Remove the "
141  "operator."),
142  node.location);
143  } else if (rightType == Type::Variable) {
144  RaiseTypeError(
145  _("Operators (+, -) can't be used in variable names. Remove "
146  "the operator from the variable name."),
147  node.location);
148  }
149  }

202  node.location);
203  }

205  // It can't happen.

208  node.location);

216  childType = ValidateFunction(node);
217  }
218  void OnVisitEmptyNode(EmptyNode& node) override {
219  ReportAnyError(node);
220  gd::String message;
221  if (parentType == Type::Number) {
222  message = _("You must enter a number or a valid expression call.");
223  } else if (parentType == Type::String) {
224  message = _(
225  "You must enter a text (between quotes) or a valid expression call.");
226  } else if (parentType == Type::Variable) {
227  message = _("You must enter a variable name.");
228  } else if (parentType == Type::Object) {
229  message = _("You must enter a valid object name.");
230  } else {
231  // It can't happen.
232  message = _("You must enter a valid expression.");
233  }
234  RaiseTypeError(message, node.location);
235  childType = Type::Empty;
236  }

242  void ReportAnyError(const ExpressionNode& node) {
243  if (node.diagnostic && node.diagnostic->IsError()) {
244  // Syntax errors are holden by the AST nodes.
245  // It's fine to give pointers on them as the AST live longer than errors
246  // handling.
247  errors.push_back(node.diagnostic.get());
248  }
249  }

251  void RaiseError(const gd::String &type,
252  const gd::String &message, const ExpressionParserLocation &location) {
253  auto diagnostic = gd::make_unique<ExpressionParserError>(
254  type, message, location);
255  errors.push_back(diagnostic.get());
256  // Errors found by the validator are not holden by the AST nodes.
257  // They must be owned by the validator to keep living while errors are
258  // handled by the caller.
259  supplementalErrors.push_back(std::move(diagnostic));
260  }

272  void CheckType(Type expect, Type actual, const ExpressionParserLocation &location) {
273  if (actual == Type::String) {
274  if (expect == Type::Number) {
275  RaiseTypeError(_("You entered a text, but a number was expected."),
276  location);
277  }
278  else if (expect != Type::String && expect != Type::NumberOrString) {
279  RaiseTypeError(
280  _("You entered a text, but this type was expected:") + " " + TypeToString(expect),
281  location);
282  }
283  }
284  else if (actual == Type::Number) {
285  if (expect == Type::String) {
286  RaiseTypeError(
287  _("You entered a number, but a text was expected (in quotes)."),
288  location);
289  }
290  else if (expect != Type::Number && expect != Type::NumberOrString) {
291  RaiseTypeError(
292  _("You entered a number, but this type was expected:") + " " + TypeToString(expect),
293  location);
294  }
295  }
