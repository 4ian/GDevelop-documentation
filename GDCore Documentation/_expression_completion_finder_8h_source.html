284  : public ExpressionParser2NodeWorker {
285  public:
290  static std::vector<ExpressionCompletionDescription>
291  GetCompletionDescriptionsFor(gd::ExpressionNode& node,
292  size_t searchedPosition) {
293  gd::ExpressionNodeLocationFinder finder(searchedPosition);
294  node.Visit(finder);
295  gd::ExpressionNode* nodeAtLocation = finder.GetNode();
296 
297  if (nodeAtLocation == nullptr) {
298  std::vector<ExpressionCompletionDescription> emptyCompletions;
299  return emptyCompletions;
300  }
301 
302  gd::ExpressionNode* maybeParentNodeAtLocation = finder.GetParentNode();
303  gd::ExpressionCompletionFinder autocompletionProvider(
304  searchedPosition, maybeParentNodeAtLocation);
305  nodeAtLocation->Visit(autocompletionProvider);
306  return autocompletionProvider.GetCompletionDescriptions();
307  }

313  GetCompletionDescriptions() {
314  return completions;
315  };

320  void OnVisitSubExpressionNode(SubExpressionNode& node) override {
321  completions.push_back(ExpressionCompletionDescription::ForObject(
322  node.type, "", searchedPosition + 1, searchedPosition + 1));
323  completions.push_back(ExpressionCompletionDescription::ForExpression(
324  node.type, "", searchedPosition + 1, searchedPosition + 1));
325  }
326  void OnVisitOperatorNode(OperatorNode& node) override {
327  completions.push_back(ExpressionCompletionDescription::ForObject(
328  node.type, "", searchedPosition + 1, searchedPosition + 1));
329  completions.push_back(ExpressionCompletionDescription::ForExpression(
330  node.type, "", searchedPosition + 1, searchedPosition + 1));
331  }
332  void OnVisitUnaryOperatorNode(UnaryOperatorNode& node) override {
333  completions.push_back(ExpressionCompletionDescription::ForObject(
334  node.type, "", searchedPosition + 1, searchedPosition + 1));
335  completions.push_back(ExpressionCompletionDescription::ForExpression(
336  node.type, "", searchedPosition + 1, searchedPosition + 1));
337  }
338  void OnVisitNumberNode(NumberNode& node) override {
339  // No completions
340  }
341  void OnVisitTextNode(TextNode& node) override {
342  // Completions are searched in the case the text node is a parameter of a
343  // function call.
344  FunctionCallNode* functionCall =
345  dynamic_cast<FunctionCallNode*>(maybeParentNodeAtLocation);
346  if (functionCall != nullptr) {
347  int parameterIndex = -1;
348  for (int i = 0; i < functionCall->parameters.size(); i++) {
349  if (functionCall->parameters.at(i).get() == &node) {
350  parameterIndex = i;
351  break;
352  }
353  }
354  if (parameterIndex < 0) {
355  return;
356  }
357  // Search the parameter metadata index skipping invisible ones.
358  int visibleParameterIndex = 0;
359  int metadataParameterIndex = 0;
360  const gd::ParameterMetadata* parameterMetadata = nullptr;
361  while (metadataParameterIndex <
362  functionCall->expressionMetadata.parameters.size()) {
363  if (!functionCall->expressionMetadata.parameters[metadataParameterIndex]
364  .IsCodeOnly()) {
365  if (visibleParameterIndex == parameterIndex) {
366  parameterMetadata = &functionCall->expressionMetadata
367  .parameters[metadataParameterIndex];
368  }
369  visibleParameterIndex++;
370  }
371  metadataParameterIndex++;
372  }
373  const int visibleParameterCount = visibleParameterIndex;
374  if (parameterMetadata == nullptr) {
375  // There are too many parameters in the expression, this text node is
376  // not actually linked to a parameter expected by the function call.
377  return;
378  }
379 
380  const gd::String& type = parameterMetadata->GetType();
381  if (type == "string") {
382  // No completions for an arbitrary string.
383  return;
384  }
385 
386  bool isLastParameter = parameterIndex == visibleParameterCount - 1;
387  completions.push_back(ExpressionCompletionDescription::ForText(
388  type,
389  *parameterMetadata,
390  node.text,
391  node.location.GetStartPosition(),
392  node.location.GetEndPosition(),
393  isLastParameter,
394  functionCall->objectName));
395  }
396  }
397  void OnVisitVariableNode(VariableNode& node) override {
398  completions.push_back(ExpressionCompletionDescription::ForVariable(
399  node.type,
400  node.name,
401  node.location.GetStartPosition(),
402  node.location.GetEndPosition(),
403  node.objectName));
404  }
405  void OnVisitVariableAccessorNode(VariableAccessorNode& node) override {
406  // No completions
407  }
408  void OnVisitVariableBracketAccessorNode(
409  VariableBracketAccessorNode& node) override {
410  // No completions
411  }
412  void OnVisitIdentifierNode(IdentifierNode& node) override {
413  if (gd::ParameterMetadata::IsObject(node.type)) {
414  // Only show completions of objects if an object is required
415  completions.push_back(ExpressionCompletionDescription::ForObject(
416  node.type,
417  node.identifierName,
418  node.location.GetStartPosition(),
419  node.location.GetEndPosition()));
420  } else {
421  // Show completions for expressions and objects otherwise.
422  completions.push_back(ExpressionCompletionDescription::ForObject(
423  node.type,
424  node.identifierName,
425  node.location.GetStartPosition(),
426  node.location.GetEndPosition()));
427  completions.push_back(ExpressionCompletionDescription::ForExpression(
428  node.type,
429  node.identifierName,
430  node.location.GetStartPosition(),
431  node.location.GetEndPosition()));
432  }
433  }
434  void OnVisitObjectFunctionNameNode(ObjectFunctionNameNode& node) override {
435  if (!node.behaviorFunctionName.empty() ||
436  node.behaviorNameNamespaceSeparatorLocation.IsValid()) {
437  // Behavior function (or behavior function being written, with the
438  // function name missing)
439  if (IsCaretOn(node.objectNameLocation)) {
440  completions.push_back(ExpressionCompletionDescription::ForObject(
441  node.type,
442  node.objectName,
443  node.objectNameLocation.GetStartPosition(),
444  node.objectNameLocation.GetEndPosition()));
445  } else if (IsCaretOn(node.objectNameDotLocation) ||
446  IsCaretOn(node.objectFunctionOrBehaviorNameLocation)) {
447  completions.push_back(ExpressionCompletionDescription::ForBehavior(
448  node.objectFunctionOrBehaviorName,
449  node.objectFunctionOrBehaviorNameLocation.GetStartPosition(),
450  node.objectFunctionOrBehaviorNameLocation.GetEndPosition(),
451  node.objectName));
452  } else if (IsCaretOn(node.behaviorNameNamespaceSeparatorLocation) ||
453  IsCaretOn(node.behaviorFunctionNameLocation)) {
454  completions.push_back(ExpressionCompletionDescription::ForExpression(
455  node.type,
456  node.behaviorFunctionName,
457  node.behaviorFunctionNameLocation.GetStartPosition(),
458  node.behaviorFunctionNameLocation.GetEndPosition(),
459  node.objectName,
460  node.objectFunctionOrBehaviorName));
461  }
462  } else {
463  // Object function or behavior name
464  if (IsCaretOn(node.objectNameLocation)) {
465  completions.push_back(ExpressionCompletionDescription::ForObject(
466  node.type,
467  node.objectName,
468  node.objectNameLocation.GetStartPosition(),
469  node.objectNameLocation.GetEndPosition()));
470  } else if (IsCaretOn(node.objectNameDotLocation) ||
471  IsCaretOn(node.objectFunctionOrBehaviorNameLocation)) {
472  completions.push_back(ExpressionCompletionDescription::ForBehavior(
473  node.objectFunctionOrBehaviorName,
474  node.objectFunctionOrBehaviorNameLocation.GetStartPosition(),
475  node.objectFunctionOrBehaviorNameLocation.GetEndPosition(),
476  node.objectName));
477  completions.push_back(ExpressionCompletionDescription::ForExpression(
478  node.type,
479  node.objectFunctionOrBehaviorName,
480  node.objectFunctionOrBehaviorNameLocation.GetStartPosition(),
481  node.objectFunctionOrBehaviorNameLocation.GetEndPosition(),
482  node.objectName));
483  }
484  }
485  }
486  void OnVisitFunctionCallNode(FunctionCallNode& node) override {
487  bool isCaretOnParenthesis = IsCaretOn(node.openingParenthesisLocation) ||
488  IsCaretOn(node.closingParenthesisLocation);
489 
490  if (!node.behaviorName.empty()) {
491  // Behavior function
492  if (IsCaretOn(node.objectNameLocation)) {
493  completions.push_back(ExpressionCompletionDescription::ForObject(
494  node.type,
495  node.objectName,
496  node.objectNameLocation.GetStartPosition(),
497  node.objectNameLocation.GetEndPosition()));
498  } else if (IsCaretOn(node.objectNameDotLocation) ||
499  IsCaretOn(node.behaviorNameLocation)) {
500  completions.push_back(ExpressionCompletionDescription::ForBehavior(
501  node.behaviorName,
502  node.behaviorNameLocation.GetStartPosition(),
503  node.behaviorNameLocation.GetEndPosition(),
504  node.objectName));
505  } else {
506  completions.push_back(ExpressionCompletionDescription::ForExpression(
507  node.type,
508  node.functionName,
509  node.functionNameLocation.GetStartPosition(),
510  node.functionNameLocation.GetEndPosition(),
511  node.objectName,
512  node.behaviorName)
513  .SetIsExact(isCaretOnParenthesis));
514  }
515  } else if (!node.objectName.empty()) {
516  // Object function
517  if (IsCaretOn(node.objectNameLocation)) {
518  completions.push_back(ExpressionCompletionDescription::ForObject(
519  node.type,
520  node.objectName,
521  node.objectNameLocation.GetStartPosition(),
522  node.objectNameLocation.GetEndPosition()));
523  } else {
524  // Add completions for behaviors, because we could imagine that the user
525  // wants to move from an object function to a behavior function, and so
526  // need behavior completions. Do this unless we're on the parenthesis
527  // (at which point we're only showing informative message about the
528  // function).
529  if (!isCaretOnParenthesis) {
530  completions.push_back(ExpressionCompletionDescription::ForBehavior(
531  node.functionName,
532  node.objectNameLocation.GetStartPosition(),
533  node.objectNameLocation.GetEndPosition(),
534  node.objectName));
535  }
536 
537  completions.push_back(ExpressionCompletionDescription::ForExpression(
538  node.type,
539  node.functionName,
540  node.functionNameLocation.GetStartPosition(),
541  node.functionNameLocation.GetEndPosition(),
542  node.objectName)
543  .SetIsExact(isCaretOnParenthesis));
544  }
545  } else {
546  // Free function
547  completions.push_back(ExpressionCompletionDescription::ForExpression(
548  node.type,
549  node.functionName,
550  node.functionNameLocation.GetStartPosition(),
551  node.functionNameLocation.GetEndPosition())
552  .SetIsExact(isCaretOnParenthesis));
553  }
554  }
555  void OnVisitEmptyNode(EmptyNode& node) override {
556  completions.push_back(ExpressionCompletionDescription::ForObject(
557  node.type,
558  node.text,
559  node.location.GetStartPosition(),
560  node.location.GetEndPosition()));
561  completions.push_back(ExpressionCompletionDescription::ForExpression(
562  node.type,
563  node.text,
564  node.location.GetStartPosition(),
565  node.location.GetEndPosition()));
566  }
567 
568  private:
569  bool IsCaretOn(const ExpressionParserLocation& location,
570  bool inclusive = false) {
571  if (!location.IsValid()) return false;
572 
573  return (location.GetStartPosition() <= searchedPosition &&
574  ((!inclusive && searchedPosition < location.GetEndPosition()) ||
575  (inclusive && searchedPosition <= location.GetEndPosition())));
576  }
577 
578  ExpressionCompletionFinder(size_t searchedPosition_,
579  gd::ExpressionNode* maybeParentNodeAtLocation_)
580  : searchedPosition(searchedPosition_),
581  maybeParentNodeAtLocation(maybeParentNodeAtLocation_){};
582 
583  std::vector<ExpressionCompletionDescription> completions;
584  size_t searchedPosition;
585  gd::ExpressionNode* maybeParentNodeAtLocation;
586 };
