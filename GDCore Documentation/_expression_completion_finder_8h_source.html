292  static std::vector<ExpressionCompletionDescription>
293  GetCompletionDescriptionsFor(gd::ExpressionNode& node,
294  size_t searchedPosition) {
295  gd::ExpressionNodeLocationFinder finder(searchedPosition);
296  node.Visit(finder);
297  gd::ExpressionNode* nodeAtLocation = finder.GetNode();
298 
299  if (nodeAtLocation == nullptr) {
300  std::vector<ExpressionCompletionDescription> emptyCompletions;
301  return emptyCompletions;
302  }
303 
304  gd::ExpressionNode* maybeParentNodeAtLocation = finder.GetParentNode();
305  gd::ExpressionCompletionFinder autocompletionProvider(
306  searchedPosition, maybeParentNodeAtLocation);
307  nodeAtLocation->Visit(autocompletionProvider);
308  return autocompletionProvider.GetCompletionDescriptions();
309  }

323  completions.push_back(ExpressionCompletionDescription::ForObject(
324  node.type, "", searchedPosition + 1, searchedPosition + 1));
325  completions.push_back(ExpressionCompletionDescription::ForExpression(
326  node.type, "", searchedPosition + 1, searchedPosition + 1));
327  }

332  completions.push_back(ExpressionCompletionDescription::ForObject(

341  void OnVisitTextNode(TextNode& node) override {
342  // Completions are searched in the case the text node is a parameter of a
343  // function call.
344  FunctionCallNode* functionCall =
345  dynamic_cast<FunctionCallNode*>(maybeParentNodeAtLocation);
346  if (functionCall != nullptr) {
347  int parameterIndex = -1;
348  for (int i = 0; i < functionCall->parameters.size(); i++) {
349  if (functionCall->parameters.at(i).get() == &node) {
350  parameterIndex = i;
351  break;
352  }
353  }
354  if (parameterIndex < 0) {
355  return;
356  }
357  // Search the parameter metadata index skipping invisible ones.
358  size_t visibleParameterIndex = 0;
359  size_t metadataParameterIndex = ExpressionParser2::WrittenParametersFirstIndex(
360  functionCall->objectName, functionCall->behaviorName);
361 
362  const gd::ParameterMetadata* parameterMetadata = nullptr;
363  while (metadataParameterIndex <
364  functionCall->expressionMetadata.parameters.size()) {
365  if (!functionCall->expressionMetadata.parameters[metadataParameterIndex]
366  .IsCodeOnly()) {
367  if (visibleParameterIndex == parameterIndex) {
368  parameterMetadata = &functionCall->expressionMetadata
369  .parameters[metadataParameterIndex];
370  }
371  visibleParameterIndex++;
372  }
373  metadataParameterIndex++;
374  }
375  const int visibleParameterCount = visibleParameterIndex;
376  if (parameterMetadata == nullptr) {
377  // There are too many parameters in the expression, this text node is
378  // not actually linked to a parameter expected by the function call.
379  return;
380  }
381 
382  const gd::String& type = parameterMetadata->GetType();
383  if (type == "string") {
384  // No completions for an arbitrary string.
385  return;
386  }
387 
388  bool isLastParameter = parameterIndex == visibleParameterCount - 1;
389  completions.push_back(ExpressionCompletionDescription::ForText(
390  type,
391  *parameterMetadata,
392  node.text,
393  node.location.GetStartPosition(),
394  node.location.GetEndPosition(),
395  isLastParameter,
396  functionCall->objectName));
397  }
398  }
399  void OnVisitVariableNode(VariableNode& node) override {
400  completions.push_back(ExpressionCompletionDescription::ForVariable(
401  node.type,
402  node.name,
403  node.location.GetStartPosition(),
404  node.location.GetEndPosition(),
405  node.objectName));
406  }
407  void OnVisitVariableAccessorNode(VariableAccessorNode& node) override {
408  // No completions
409  }
410  void OnVisitVariableBracketAccessorNode(
411  VariableBracketAccessorNode& node) override {
412  // No completions
413  }
414  void OnVisitIdentifierNode(IdentifierNode& node) override {
415  if (gd::ParameterMetadata::IsObject(node.type)) {
416  // Only show completions of objects if an object is required
417  completions.push_back(ExpressionCompletionDescription::ForObject(
418  node.type,
419  node.identifierName,
420  node.location.GetStartPosition(),
421  node.location.GetEndPosition()));
422  } else {
423  // Show completions for expressions and objects otherwise.
424  completions.push_back(ExpressionCompletionDescription::ForObject(
425  node.type,
426  node.identifierName,
427  node.location.GetStartPosition(),
428  node.location.GetEndPosition()));
429  completions.push_back(ExpressionCompletionDescription::ForExpression(
430  node.type,
431  node.identifierName,
432  node.location.GetStartPosition(),
433  node.location.GetEndPosition()));
434  }
435  }
436  void OnVisitObjectFunctionNameNode(ObjectFunctionNameNode& node) override {
437  if (!node.behaviorFunctionName.empty() ||
438  node.behaviorNameNamespaceSeparatorLocation.IsValid()) {
439  // Behavior function (or behavior function being written, with the
440  // function name missing)
441  if (IsCaretOn(node.objectNameLocation)) {
442  completions.push_back(ExpressionCompletionDescription::ForObject(
443  node.type,
444  node.objectName,
445  node.objectNameLocation.GetStartPosition(),
446  node.objectNameLocation.GetEndPosition()));
447  } else if (IsCaretOn(node.objectNameDotLocation) ||
448  IsCaretOn(node.objectFunctionOrBehaviorNameLocation)) {
449  completions.push_back(ExpressionCompletionDescription::ForBehavior(
450  node.objectFunctionOrBehaviorName,
451  node.objectFunctionOrBehaviorNameLocation.GetStartPosition(),
452  node.objectFunctionOrBehaviorNameLocation.GetEndPosition(),
453  node.objectName));
454  } else if (IsCaretOn(node.behaviorNameNamespaceSeparatorLocation) ||
455  IsCaretOn(node.behaviorFunctionNameLocation)) {
456  completions.push_back(ExpressionCompletionDescription::ForExpression(
457  node.type,
458  node.behaviorFunctionName,
459  node.behaviorFunctionNameLocation.GetStartPosition(),
460  node.behaviorFunctionNameLocation.GetEndPosition(),
461  node.objectName,
462  node.objectFunctionOrBehaviorName));
463  }
464  } else {
465  // Object function or behavior name
466  if (IsCaretOn(node.objectNameLocation)) {
467  completions.push_back(ExpressionCompletionDescription::ForObject(
468  node.type,
469  node.objectName,
470  node.objectNameLocation.GetStartPosition(),
471  node.objectNameLocation.GetEndPosition()));
472  } else if (IsCaretOn(node.objectNameDotLocation) ||
473  IsCaretOn(node.objectFunctionOrBehaviorNameLocation)) {
474  completions.push_back(ExpressionCompletionDescription::ForBehavior(
475  node.objectFunctionOrBehaviorName,
476  node.objectFunctionOrBehaviorNameLocation.GetStartPosition(),
477  node.objectFunctionOrBehaviorNameLocation.GetEndPosition(),
478  node.objectName));
479  completions.push_back(ExpressionCompletionDescription::ForExpression(
480  node.type,
481  node.objectFunctionOrBehaviorName,
482  node.objectFunctionOrBehaviorNameLocation.GetStartPosition(),
483  node.objectFunctionOrBehaviorNameLocation.GetEndPosition(),
484  node.objectName));
485  }
486  }
487  }
488  void OnVisitFunctionCallNode(FunctionCallNode& node) override {
489  bool isCaretOnParenthesis = IsCaretOn(node.openingParenthesisLocation) ||
490  IsCaretOn(node.closingParenthesisLocation);
491 
492  if (!node.behaviorName.empty()) {
493  // Behavior function
494  if (IsCaretOn(node.objectNameLocation)) {
495  completions.push_back(ExpressionCompletionDescription::ForObject(
496  node.type,
497  node.objectName,
498  node.objectNameLocation.GetStartPosition(),
499  node.objectNameLocation.GetEndPosition()));
500  } else if (IsCaretOn(node.objectNameDotLocation) ||
501  IsCaretOn(node.behaviorNameLocation)) {
502  completions.push_back(ExpressionCompletionDescription::ForBehavior(
503  node.behaviorName,
504  node.behaviorNameLocation.GetStartPosition(),
505  node.behaviorNameLocation.GetEndPosition(),
506  node.objectName));
507  } else {
508  completions.push_back(ExpressionCompletionDescription::ForExpression(
509  node.type,
510  node.functionName,
511  node.functionNameLocation.GetStartPosition(),
512  node.functionNameLocation.GetEndPosition(),
513  node.objectName,
514  node.behaviorName)
515  .SetIsExact(isCaretOnParenthesis));
516  }
517  } else if (!node.objectName.empty()) {
518  // Object function
519  if (IsCaretOn(node.objectNameLocation)) {
520  completions.push_back(ExpressionCompletionDescription::ForObject(
521  node.type,
522  node.objectName,
523  node.objectNameLocation.GetStartPosition(),
524  node.objectNameLocation.GetEndPosition()));
525  } else {
526  // Add completions for behaviors, because we could imagine that the user
527  // wants to move from an object function to a behavior function, and so
528  // need behavior completions. Do this unless we're on the parenthesis
529  // (at which point we're only showing informative message about the
530  // function).
531  if (!isCaretOnParenthesis) {
532  completions.push_back(ExpressionCompletionDescription::ForBehavior(
533  node.functionName,
534  node.objectNameLocation.GetStartPosition(),
535  node.objectNameLocation.GetEndPosition(),
536  node.objectName));
537  }
538 
539  completions.push_back(ExpressionCompletionDescription::ForExpression(
540  node.type,
541  node.functionName,
542  node.functionNameLocation.GetStartPosition(),
543  node.functionNameLocation.GetEndPosition(),
544  node.objectName)
545  .SetIsExact(isCaretOnParenthesis));
546  }
547  } else {
548  // Free function
549  completions.push_back(ExpressionCompletionDescription::ForExpression(
550  node.type,
551  node.functionName,
552  node.functionNameLocation.GetStartPosition(),
553  node.functionNameLocation.GetEndPosition())
554  .SetIsExact(isCaretOnParenthesis));
555  }
556  }
557  void OnVisitEmptyNode(EmptyNode& node) override {
558  completions.push_back(ExpressionCompletionDescription::ForObject(
559  node.type,
560  node.text,
561  node.location.GetStartPosition(),
562  node.location.GetEndPosition()));
563  completions.push_back(ExpressionCompletionDescription::ForExpression(
564  node.type,
565  node.text,
566  node.location.GetStartPosition(),
567  node.location.GetEndPosition()));
568  }
569 
570  private:
571  bool IsCaretOn(const ExpressionParserLocation& location,
572  bool inclusive = false) {
573  if (!location.IsValid()) return false;
574 
575  return (location.GetStartPosition() <= searchedPosition &&
576  ((!inclusive && searchedPosition < location.GetEndPosition()) ||
577  (inclusive && searchedPosition <= location.GetEndPosition())));
578  }
579 
580  ExpressionCompletionFinder(size_t searchedPosition_,
581  gd::ExpressionNode* maybeParentNodeAtLocation_)
582  : searchedPosition(searchedPosition_),
583  maybeParentNodeAtLocation(maybeParentNodeAtLocation_){};
584 
585  std::vector<ExpressionCompletionDescription> completions;
586  size_t searchedPosition;
587  gd::ExpressionNode* maybeParentNodeAtLocation;
588 };
