42  bool negatePredicate,
43  Pred predicate) {
44  bool isTrue = false;
45 
46  // Create a boolean for each object
47  std::vector<std::vector<bool> > pickedList;
48  for (RuntimeObjectsLists::const_iterator it = pickedObjectsLists.begin();
49  it != pickedObjectsLists.end();
50  ++it) {
51  std::vector<bool> arr;
52  arr.assign(it->second->size(), false);
53  pickedList.push_back(arr);
54  }
55 
56  // Pick objects which are fulfulling the predicate.
57  std::size_t i = 0;
58  for (RuntimeObjectsLists::const_iterator it = pickedObjectsLists.begin();
59  it != pickedObjectsLists.end();
60  ++it, ++i) {
61  if (!it->second) continue;
62  const std::vector<RuntimeObject *> &arr1 = *it->second;
63 
64  for (std::size_t k = 0; k < arr1.size(); ++k) {
65  if (negatePredicate ^ predicate(arr1[k])) {
66  pickedList[i][k] = true;
67  isTrue = true;
68  }
69  }
70  }
71 
72  // Trim not picked objects from lists.
73  i = 0;
74  for (RuntimeObjectsLists::const_iterator it = pickedObjectsLists.begin();
75  it != pickedObjectsLists.end();
76  ++it, ++i) {
77  size_t finalSize = 0;
78  if (!it->second) continue;
79  std::vector<RuntimeObject *> &arr = *it->second;
80 
81  for (std::size_t k = 0; k < arr.size(); ++k) {
82  RuntimeObject *obj = arr[k];
83  if (pickedList[i][k]) {
84  arr[finalSize] = obj;
85  finalSize++;
86  }
