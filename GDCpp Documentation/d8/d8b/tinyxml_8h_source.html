335  // Get a character, while interpreting entities.
336  // The length can be from 0 to 4 bytes.
337  inline static const char* GetChar( const char* p, char* _value, int* length, TiXmlEncoding encoding )
338  {
339  assert( p );
340  if ( encoding == TIXML_ENCODING_UTF8 )
341  {
342  *length = utf8ByteTable[ *((const unsigned char*)p) ];
343  assert( *length >= 0 && *length < 5 );
344  }
345  else
346  {
347  *length = 1;
348  }
349 
350  if ( *length == 1 )
351  {
352  if ( *p == '&' )
353  return GetEntity( p, _value, length, encoding );
354  *_value = *p;
355  return p+1;
356  }
357  else if ( *length )
358  {
359  //strncpy( _value, p, *length ); // lots of compilers don't like this function (unsafe),
360  // and the null terminator isn't needed
361  for( int i=0; p[i] && i<*length; ++i ) {
362  _value[i] = p[i];
363  }
364  return p + (*length);
365  }
366  else
367  {
368  // Not valid text.
369  return 0;
370  }
371  }

901 /* A class used to manage a group of attributes.
902  It is only used internally, both by the ELEMENT and the DECLARATION.
903 
904  The set can be changed transparent to the Element and Declaration
905  classes that use it, but NOT transparent to the Attribute
906  which has to implement a next() and previous() method. Which makes
907  it a bit problematic and prevents the use of STL.

910  - I like circular lists
911  - it demonstrates some independence from the (typical) doubly linked list.
912 */

914 {
915 public:
916  TiXmlAttributeSet();
